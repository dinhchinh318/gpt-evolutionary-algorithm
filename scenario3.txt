ea.experiment(
  runs = (randomGenerator = (seed = [1:1:30]) * [m.defaultRG()])
  * (problem = $problems)
  * (solver = $solvers)
  * [ea.run(name = "scenario3")];
  listeners = [
    ea.l.console(
      eFunctions = [
        f.mapValue(
          key = "avg.dist";
          of = ea.f.behaviorQuality(of = ea.f.quality(of = ea.f.best()));
          format = "%8.5f"
        )
      ];
      onlyLast = false
    );
    ea.l.csv(
      path = "./out/scenario3-progress.csv";
      defaultKFunctions = [];
      kFunctions = [
        f.mappableKey(name = "seed"; key = "randomGenerator.seed");
        f.mappableKey(name = "problem"; key = "problem.name");
        f.mappableKey(name = "solver_sigma"; key = "solver.name")
      ];
      eFunctions = [
        f.size(
          of = ea.f.genotype(of = ea.f.best());
          format = "%d"
        );
        f.mapValue(
          key = "avg.dist";
          of = ea.f.behaviorQuality(of = ea.f.quality(of = ea.f.best()));
          format = "%.17f"
        )
      ];
      onlyLast = false
    );
    ea.l.csv(
      path = "./out/scenario3-final.csv";
      defaultKFunctions = [];
      kFunctions = [
